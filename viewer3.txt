    <%
        // JDBC URL, username, and password of the MySQL server
        String jdbcUrl = "jdbc:mysql://192.168.0.65:3306/upgw";
        String dbUsername = "root";
        String dbPassword = "happycoding";

        // JDBC variables for opening and managing connection
        Connection conn = null;
        Statement stmt = null;
        ResultSet rs = null;

        try {
            // Load MySQL JDBC Driver
            Class.forName("com.mysql.jdbc.Driver"); // Updated driver class name for MySQL Connector/J 4.x

            // Open a connection
            conn = DriverManager.getConnection(jdbcUrl, dbUsername, dbPassword);

            // Get the SQL command from the POST request
            String sql = request.getParameter("sqlCommand");
            if (sql == null || sql.trim().isEmpty()) {
                out.println("No SQL command provided.");
                return;
            }

            // Execute SQL query
            stmt = conn.createStatement();
            boolean isResultSet = stmt.execute(sql);

            if (isResultSet) {
                // Process the result set
                rs = stmt.getResultSet();
                int columnCount = rs.getMetaData().getColumnCount();

                // Print column names
                for (int i = 1; i <= columnCount; i++) {
                    out.print(rs.getMetaData().getColumnName(i) + "\t");
                }
                out.println("<br>");

                // Print rows
                while (rs.next()) {
                    for (int i = 1; i <= columnCount; i++) {
                        out.print(rs.getString(i) + "\t");
                    }
                    out.println("<br>");
                }
            } else {
                // Print update count
                int updateCount = stmt.getUpdateCount();
                out.println("Update Count: " + updateCount);
            }
        } catch (Exception e) {
            // Convert JspWriter to PrintWriter
            StringWriter sw = new StringWriter();
            PrintWriter pw = new PrintWriter(sw);
            e.printStackTrace(pw);
            out.println(sw.toString());
        } finally {
            // Clean-up environment
            try {
                if (rs != null) rs.close();
                if (stmt != null) stmt.close();
                if (conn != null) conn.close();
            } catch (Exception e) {
                StringWriter sw = new StringWriter();
                PrintWriter pw = new PrintWriter(sw);
                e.printStackTrace(pw);
                out.println(sw.toString());
            }
        }
    %>
